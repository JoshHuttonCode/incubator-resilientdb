50void DuckDB::CreateDB(const DuckDBInfo& config) {
51 std::string db_path = config.path();
52
53 duckdb::DBConfig db_config;
54
55 if (config.has_max_memory()) {
56 db_config.SetOption("max_memory", duckdb::Value(config.max_memory()));
57 }
58
59 LOG(INFO) << "Initializing DuckDB at "
60 << (db_path.empty() ? "in-memory" : db_path)
61 << " (autoload/autoinstall extensions enabled)";
62
63 if (db_path.empty()) {
64 db_ = std::make_unique<duckdb::DuckDB>(nullptr, &db_config);
65 } else {
66 db_ = std::make_unique<duckdb::DuckDB>(db_path, &db_config);
67 }
68
69 // Ensure extension auto-install/load is enabled at the connection level too.
70 try {
71 duckdb::Connection init_conn(*db_);
72 init_conn.Query("SET autoload_known_extensions=1");
73 init_conn.Query("SET autoinstall_known_extensions=1");
74 } catch (const std::exception& e) {
75 LOG(ERROR) << "Failed to set DuckDB extension auto-load/install flags: "
76 << e.what();
77 }
78}

80std::string DuckDB::ExecuteSQL(const std::string& sql_string){
81 if (sql_string.empty()) {
82 return "Error: empty SQL query";
83 }
84
85 if (!db_) {
86 LOG(ERROR) << "DuckDB is not initialized";
87 return "Error: database not initialized";
88 }
89
90 try {
91 LOG(INFO) << "Executing SQL: " << sql_string;
92 conn_ = std::make_unique<duckdb::Connection> (*db_);
93 auto result = conn_->Query(sql_string);
94
95 if (!result) {
96 LOG(ERROR) << "Query returned nullptr";
97 return "Error: query returned no result";
98 }
99
100 if (result->HasError()) {
101 LOG(ERROR) << "SQL Error: " << result->GetError();
102 return "Error: " + result->GetError();
103 }
104
105 std::string response = result->ToString();
106 LOG(INFO) << "SQL succeeded. Rows: " << result->RowCount()
107 << " Cols: " << result->ColumnCount();
108 LOG(INFO) << "SQL Result:\n" << response;
109 return response;
110 } catch (const std::exception& e) {
111 LOG(ERROR) << "SQL execution threw: " << e.what();
112 return std::string("Error: ") + e.what();
113 }
114}
